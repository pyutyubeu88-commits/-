# 확장 설명 (첫 번째 줄) # 파일 위치: .conway/extensions/{이름}.cnw.py SYSTEM_PROMPT = """ 이 확장은 특정 도메인에 특화된 지시사항을 제공합니다. 예: GitHub PR 리뷰, 코드 분석, 데이터 처리 등 """ def on_task(task): """작업 전처리 로직 (선택)""" return task
